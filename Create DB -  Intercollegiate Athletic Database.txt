CREATE TABLE CUSTOMER 
( 
CUSTNO VARCHAR2(8) NOT NULL, 
CUSTNAME VARCHAR2(50) NOT NULL, 
ADDRESS VARCHAR2(50) NOT NULL, 
INTERNAL CHAR(1) NOT NULL, 
CONTACT VARCHAR2(50) NOT NULL, 
PHONE VARCHAR2(11) NOT NULL, 
CITY VARCHAR2(50) NOT NULL, 
STATE VARCHAR2(4) NOT NULL, 
ZIP VARCHAR2(10) NOT NULL, 
CONSTRAINT CUSTNO_PK PRIMARY KEY (CUSTNO) 
);

CREATE TABLE EMPLOYEE 
( 
EMPNO VARCHAR2(8), 
EMPNAME VARCHAR2(50) NOT NULL, 
DEPARTMENT VARCHAR2(50) NOT NULL, 
EMAIL VARCHAR2(50) NOT NULL, 
PHONE VARCHAR2(11) NOT NULL, 
CONSTRAINT EMPNO_PK PRIMARY KEY (EMPNO) 
);

CREATE TABLE FACILITY 
( 
FACNO VARCHAR2(8) NOT NULL, 
FACNAME VARCHAR2(50) NOT NULL, 
CONSTRAINT FACNO_PK PRIMARY KEY (FACNO) 
);

CREATE TABLE LOCATION 
( 
LOCNO VARCHAR2(8) NOT NULL, 
FACNO VARCHAR2(8) NOT NULL, 
LOCNAME VARCHAR2(50) NOT NULL, 
CONSTRAINT LOCNO_PK PRIMARY KEY (LOCNO), 
CONSTRAINT FACNO_FK FOREIGN KEY (FACNO) 
);

CREATE TABLE RESOURCETBL 
( 
RESNO VARCHAR2(8) NOT NULL, 
RESNAME VARCHAR2(50) NOT NULL, 
RATE DECIMAL(10,2) NOT NULL, 
CONSTRAINT RESNO_PK PRIMARY KEY (RESNO), 
CONSTRAINT RATE_CK CHECK (RATE > 0) 
);

CREATE TABLE EVENTREQUEST 
( 
EVENTNO VARCHAR2(8) NOT NULL, 
DATEHELD DATE,  
DATEREQ DATE, 
FACNO VARCHAR2(8) NOT NULL, 
CUSTNO VARCHAR2(8) NOT NULL, 
DATEAUTH DATE 
STATUS VARCHAR2(12) NOT NULL, 
ESTCOST DECIMAL(10,2) NOT NULL, 
ESTAUDIENCE INTEGER(10) NOT NULL, 
BUDNO VARCHAR2(12), 
CONSTRAINT EVENTNO_PK PRIMARY KEY (EVENTNO), 
CONSTRAINT FACNO_FK FOREIGN KEY (FACNO) REFERENCES FACILITY , 
CONSTRAINT CUSTNO_FK FOREIGN KEY (CUSTNO)REFERENCES CUSTOMER, 
CONSTRAINT STATUS_CK CHECK (STATUS IN ('PENDING','DENIED','APPROVED')), 
CONSTRAINT ESTAUDIENCE_CK CHECK (ESTAUDIENCE > 0) 
);

CREATE TABLE EVENTPLAN 
( 
PLANNO VARCHAR2(8) NOT NULL, 
EVENTNO VARCHAR2(8) NOT NULL, 
WORKDATE DATE, 
NOTES VARCHAR2(50), 
ACTIVITY VARCHAR2(50) NOT NULL, 
EMPNO VARCHAR2(8), 
CONSTRAINT PLANNO_PK PRIMARY KEY (PLANNO), 
CONSTRAINT EVENTNO_FK FOREIGN KEY (EVENTNO) REFERENCES EVENTREQUEST, 
CONSTRAINT EMPNO_FK FOREIGN KEY (EMPNO) REFERENCES EMPLOYEE 
);

CREATE TABLE EVENTPLANLINE 
( 
PLANNO VARCHAR2(8) NOT NULL, 
LINENO INTEGER(12) NOT NULL, 
TIMESTART DATE, 
TIMEEND DATE, 
NUMBERFLD INTEGER(12) NOT NULL, 
LOCNO VARCHAR2(8) NOT NULL, 
RESNO VARCHAR2(8) NOT NULL 
CONSTRAINT LINENO_PK PRIMARY KEY (LINENO), 
CONSTRAINT PLANNO_FK FOREIGN KEY (PLANNO) REFERENCES EVENTPLAN, 
CONSTRAINT LOCNO_FK FOREIGN KEY (LOCNO) REFERENCES LOCATION, 
CONSTRAINT RESNO_FK FOREIGN KEY (RESNO) REFERENCES RESOURCETBL, 
CONSTRAINT TIME_CK CHECK (TIMESTART < TIMEEND) 
);
